# Setting up global options
library(knitr)
opts_chunk$set(
  warning = FALSE,
  message = FALSE,
  fig.align = "center"
)

# Loading the dataset
load("../Data/PFCK_penile.Rdata")
source("../R/nice_tables.R")
source("../R/gplots.R")

# Defining the predictor variable and label
pred <- DATA$cd8ki67_tumor_pos
pred_label <- "CD8/Ki67 expression in tumor lymphocytes"

Outcome analysis

# Defining status, follow-up and plot title
FU <- DATA$fu_last_mo
status <- pred

# Defining predictors and labels
predictors <- list(status)
varlabels <- pred_label

# Survival - Overall mortality
outcome <- DATA$dead
leg_pos <- c(0.875, 0.875)
gsurv(FU, outcome, status, legend_position = leg_pos) +
  ggtitle(paste("Overall mortality by \n", pred_label))

# Survival - Cancer-related mortality
outcome <- DATA$died_disease
gsurv(FU, outcome, status, yaxis = 0.7, legend_position = leg_pos) +
  ggtitle(paste("Cancer-related mortality by \n", pred_label))
