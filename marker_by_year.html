# Setting up global options
library(knitr)
opts_chunk$set(
  warning = FALSE,
  message = FALSE,
  fig.align = "center",
  fig.width = 9
)

# Loading the dataset
load("../Data/PFCK_penile.Rdata")
source("../R/gplots.R")
source("../R/co.R")

# Preparing DATA_FULL for analysis by year
library(dplyr)
DATA_Y <- DATA_FULL %>%
  filter(tissue == "Penis-Carcinoma") %>%
  select(-c(
    pdl1_tumor_location,
    host_response,
    foxp3_lymph_intensity,
    foxp3_tumor_intensity)
  )

# Creating the year variable in the DATA_FULL dataset
library(stringr)
DATA_Y$year <- str_extract_all(DATA_Y$sp, "S\\d\\d", simplify = TRUE)
DATA_Y$year <- str_replace_all(DATA_Y$year, "S", "")
DATA_Y$year <- as.numeric(DATA_Y$year)

DATA_Y$year <- ifelse(
  DATA_Y$year > 16,
  paste("19", DATA_Y$year, sep = ""),
    ifelse (DATA_Y$year >= 0 & DATA_Y$year < 10,
      paste("200", DATA_Y$year, sep = ""),
      paste("20", DATA_Y$year, sep = ""))
)

DATA_Y$year <- DATA_Y$year %>%
  sort() %>%
  factor(ordered = TRUE)

# Creating the variable in the DATA dataset
DATA$year <- str_extract_all(DATA$sp, "S\\d\\d", simplify = TRUE)
DATA$year <- str_replace_all(DATA$year, "S", "")
DATA$year <- as.numeric(DATA$year)

DATA$year <- ifelse(
  DATA$year > 16,
  paste("19", DATA$year, sep = ""),
    ifelse (DATA$year >= 0 & DATA$year < 10,
      paste("200", DATA$year, sep = ""),
      paste("20", DATA$year, sep = ""))
)

DATA$year <- DATA$year %>%
  sort() %>%
  factor(ordered = TRUE)
