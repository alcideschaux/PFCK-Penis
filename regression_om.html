# Setting up global options
library(knitr)
opts_chunk$set(
  warning = FALSE,
  message = FALSE,
  fig.align = "center"
)

# Loading the dataset
load("../Data/PFCK_penile.Rdata")
source("../R/nice_tables.R")
library(dplyr)

# Defining the common parameters for the regression table
outcome <- DATA$dead
FU <- DATA$fu_last_mo

# Clinical predictors
clinical <- list(
  "Older age group" = DATA$age_group,
  "Caucasian race" = DATA$caucasian
)

# Pathologic predictors
pathologic <- list(
  "High histologic grade" = DATA$high_grade,
  "Tumor thickness >10 mm" = DATA$deep_tumor,
  "Lymphovascular invasion" = DATA$lvi,
  "Perineural invasion" = DATA$pni,
  "Urethral invasion" = DATA$urethra,
  "Invasion of surgical margins" = DATA$margins,
  "Inguinal lymph node metastasis" = DATA$nodes,
  "High pT (>pT1) stage" = DATA$pt,
  "High pN (pN+) stage" = DATA$pn
)

# Marker predictors
marker<- list(
  "PDL1 positivity in tumor cells" = DATA$pdl1_tumor_pos0,
  "PDL1 positivity in stromal cells" = DATA$pdl1_stroma_pos0,
  "FOXP3 positivity in tumor lymphocytes" = DATA$foxp3_tumor_pos,
  "FOXP3 positivity in stromal lymphocytes" = DATA$foxp3_stroma_pos,
  "CD8 positivity in tumor lymphocytes" = DATA$cd8_tumor_pos,
  "CD8 positivity in stromal lymphocytes" = DATA$cd8_stroma_pos,
  "CD8/Ki67 positivity in tumor lymphocytes" = DATA$cd8ki67_tumor_pos,
  "CD8/Ki67 positivity in stromal lymphocytes" = DATA$cd8ki67_stroma_pos,
  "CD8-to-FOXP3 balance in tumor lymphocytes" = DATA$cd8_foxp3_tumor,
  "CD8-to-FOXP3 balance in stromal lymphocytes" = DATA$cd8_foxp3_stroma
)
